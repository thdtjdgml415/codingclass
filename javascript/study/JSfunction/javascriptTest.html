18. 결괏값을 작성하시오.


//arr은 배열을 받고 divisor은 2를 받습니다. answer라는 변수를 배열로 선언하고
// div라는 변수에 filter라는 메서드를 사용합니다. filter메서드는 조건에 맞는
//요소들을 모아서 새로운 배열로 만듭니다. 조건은 el을 divisor =2 로 나눴을때 
//나머지가 0인 (짝수)일 경우를 모아서 새로운 배열을 만들어 div에 대입합니다.
//div에는 [2,4]가 존재하고 삼항 연산자를 사용하여 div.length(=2)가 0보다 크기 때문에
//true여서 앞쪽 조건을 적용합니다. sort는 조건에 맞게 정렬합니다. 
// 답 : 2
{
    function solution (arr,divisor) {
        var answer = [];

        const div = arr.filter(el => el% divisor == 0 );
        answer = div.length > 0 ? div.sort((a,b) => a-b): [-1];

        return answer;
    }
    document.write(solution([1,2,3,4,5],2))
}
