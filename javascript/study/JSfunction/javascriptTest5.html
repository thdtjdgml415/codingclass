//이 함수는 매개변수로 받는 수를 1부터 매개변수까지 더하는 공식이다.
//5을 매개변수로 받았기 때문에 1부터 5까지 더한다.
//답은 1+2+3+4+5 = 15이다.
function solution(n) {
    let answer = 0;
    for (let i = 1; i < n; i++) {
        answer = answer + i;
    }
    return answer;
}

// 홀수의 합과 최솟값을 구하는 함수 입니다.

// 홀수인 수를 sum변수에 계속 더해줍니다. 또한 for of문으로 계속 탐색하면서 
// 두번째 if문으로 이전 수와 비교하면 최솟값을 구합니다. push는 계속 데이터를 추가해주는 메서드 입니다.
//답은 [256,41] 입니다.
function solution(a) {
    let answer = [];
    let sum = 0, min = 1000;
    for (let x of a) {
        if (x % 2 === 1) {
            sum += x;
            if (x < min) min = x;
        }

    answer.push(min);

//이 함수는 조건에 맞는 데이터를 배열에서 추출해서 추출한 데이터를 제외하고 출력하는 함수 입니다.
//reduce는 충첩해서 더하는 함수로 변수 sum에 배열에 있는 데이터를 모두 더한 값을 대입합니다.
//sum은 140입니다. 이중 반복문을 사용해서 answer배열에서 더해서 40이 나오는 두 수를 찾습니다.
//splice는 배열에서 원하는 수를 추출하는 메서드 입니다. 15와 25를 추출하고 남은 나머지 배열을 출력하는 함수 입니다.
function solution(arr) {
    let answer = arr;
    let sum = answer.reduce((a, b) => a + b, 0);
    for (let i = 0; i < 8; i++) {
        for (let j = i + 1; j < 9; j++) {
            if ((sum - (answer[i] + answer[j])) == 100) {
                answer.splice(j, 1);
                answer.splice(i, 1);
            }
        }
    }
    return answer;
}
let arr = [20, 7, 23, 19, 10, 15, 25, 8, 13];
console.log(solution(arr));

//filter 메서드는  주어진 함수를 통과하는 모든 요소를 모아 새로운 배열을 반환합니다.
//매개변수 s에 있는 배열의 요소들을 filter메서드를 사용하여 s에 있는 배열들 중 중복하거나
//하나만 있는 요소들을 반환하는 조건을 걸고 return하였습니다. 
//답 ['good', 'time', 'student'] 
function solution(s) {
    let answer;
    answer = s.filter(function (v,i) {
        return s.indexOf(v) ===i;
    });
    return answer;
}
let str = ["good", "time", "good", "time", "student"];
console.log(solution(str));
