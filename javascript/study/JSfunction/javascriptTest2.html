01. 결괏값을 작성하시오.


//이 함수는 1996을 매개변수로 받는 q1 함수입니다. while문에 1996이 들어가고  1996을 10으로 나눈 나머지는
// answer변수에 저장되고 1996을 10으로 나눈 몫은 또 10으로 나눠지고 나머지가 answer에 더해집니다 이런식으로
//계속 더해져서 답 : 25

function q1(num) {
    let answer = 0;
    while(num) {
        answer += num % 10;
        num = parseInt(num / 10);
    }
    return answer;
}

console.log(q1(1996));
