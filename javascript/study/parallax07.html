07. 패럴랙스 이펙트 - 리빌효과

페럴랙스 효과는 스크롤을 내리면서 알맞은 스크롤의 위치에서 효과가 생기는 것을 말합니다.

javascript

이번 패럴랙스는 글자 하나씩 애니메이션에 딜레이를 줘서 출력되도록 하였습니다.
자바스크립트로 진행하였습니다. html에 reveal이라는 class가 붙은 마크업에 자식태그로 글씨가 있다면
텍스트에 span태그를 붙여서 css로 애니메이션을 줄 수 있도록 만들었습니다.

javascript 
   //reveal 클래스 자식에 글씨만 있다면, 가상으로 span태그를 만들어서 넣어주세요
const reveal = document.querySelectorAll(".reveal");

    reveal.forEach((el) => {
        let text = el.innerText;
        if (text) {
            el.innerHTML = "" + text + "";
        }
    });
    function scroll() {
        let scrollTop = window.scrollY;
    reveal.forEach(el => {

        let revealOffset = el.offsetTop + el.parentElement.offsetTop;
        let revealDelay = el.dataset.delay;

        // 딜레이 주기
        if (scrollTop > revealOffset - window.innerHeight) {
            if (!revealDelay == undefined) {
                el.classList.add("show");
            } else {
                setTimeout(() => {
                    el.classList.add("show");
                }, revealDelay)
            }
        }

        // if(scrollTop > revealOffset - window.innerHeight/2) {
        //     el.classList.add("show");
        // }
    })
    document.querySelector("#parallax__info .scroll span").innerText = Math.round(scrollTop);
    requestAnimationFrame(scroll);
}
